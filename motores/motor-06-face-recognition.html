Componente: FaceMilvus ​

Wrapper em app/engines/face_matching.py que encapsula todas as operacoes Milvus:

python
class FaceMilvus:
    COLLECTION = "face_embeddings"
    DIM = 512

    def connect(self, host: str, port: int) -> None:
        """Conecta ao Milvus Standalone e garante colecao HNSW."""
        connections.connect(host=host, port=port)
        if not utility.has_collection(self.COLLECTION):
            schema = CollectionSchema([
                FieldSchema("id", DataType.INT64, is_primary=True, auto_id=True),
                FieldSchema("creator_id", DataType.INT64),
                FieldSchema("embedding", DataType.FLOAT_VECTOR, dim=self.DIM),
            ])
            col = Collection(self.COLLECTION, schema)
            col.create_index("embedding", {
                "metric_type": "COSINE",
                "index_type": "HNSW",
                "params": {"M": 16, "efConstruction": 200},
            })

    def insert(self, creator_id: int, embedding: list[float]) -> int:
        """Insere embedding; retorna milvus_id."""
        col = Collection(self.COLLECTION)
        result = col.insert([[creator_id], [embedding]])
        col.flush()
        return result.primary_keys[0]

    def search(self, embedding: list[float], top_k: int = 5) -> list[dict]:
        """Busca HNSW COSINE. Retorna matches acima de COSINE_THRESHOLD."""
        col = Collection(self.COLLECTION)
        col.load()
        results = col.search(
            [embedding], "embedding",
            {"metric_type": "COSINE", "params": {"ef": 64}},
            limit=top_k,
            output_fields=["creator_id"],
        )
        return [
            {"creator_id": hit.entity.get("creator_id"), "score": hit.score}
            for hit in results[0]
            if hit.score >= COSINE_THRESHOLD
        ]

    def delete_by_user(self, creator_id: int) -> None:
        """LGPD: remove todos os embeddings do usuario do Milvus."""
        col = Collection(self.COLLECTION)
        col.delete(f"creator_id == {creator_id}")
        col.flush()

def sha1_hex(data: bytes) -> str:
    """SHA-1 hex para identificar imagens sem armazenar conteudo."""
    return hashlib.sha1(data).hexdigest()

def sanitize_image(image_bytes: bytes, max_size: int = 10 * 1024 * 1024) -> bytes:
    """Valida formato, tamanho e remove EXIF. Levanta ValueError se invalida."""
    if len(image_bytes) > max_size:
        raise ValueError("Imagem excede 10MB")
    img = Image.open(io.BytesIO(image_bytes))
    img.verify()
    # Remove EXIF/metadata via re-encode
    buf = io.BytesIO()
    img = Image.open(io.BytesIO(image_bytes)).convert("RGB")
    img.save(buf, format="JPEG", quality=95)
    return buf.getvalue()

def compare_batch_phash(hashes: list[str], target: str) -> list[tuple[str, int]]:
    """Compara lista de pHashes contra target. Retorna (hash, distance)."""
    target_arr = np.frombuffer(bytes.fromhex(target), dtype=np.uint8)
    results = []
    for h in hashes:
        arr = np.frombuffer(bytes.fromhex(h), dtype=np.uint8)
        dist = int(np.unpackbits(np.bitwise_xor(arr, target_arr)).sum())
        results.append((h, dist))
    return results

Ciframento de embeddings (LGPD Art. 11) ​

Dados biometricos requerem consentimento especifico + criptografia:

Chave derivada = PBKDF2-HMAC-SHA256(
    password = SECRET_KEY + str(user_id),
    salt     = user_id.to_bytes(8, 'big'),
    iterations = 100_000,
    length   = 32  # 256-bit
)
Embedding cifrado = AES-256-GCM(chave_derivada, embedding_bytes)
  • Cada usuario tem chave derivada diferente (isolamento total)
  • del img_bytes imediatamente apos gerar embedding (memoria limpa)
  • Embedding cifrado no PostgreSQL; milvus_id referencia o vetor bruto no Milvus
  • Sem armazenamento de genero, idade ou landmarks faciais

Seguranca & LGPD ​

  • Dado biometrico — requer consentimento especifico (LGPD Art. 11)
  • Embedding cifrado com AES-256-GCM (chave derivada por usuario via PBKDF2)
  • del img_bytes imediatamente apos gerar embedding — selfie nunca armazenada
  • Sem armazenamento de genero, idade ou landmarks faciais
  • Exclusao de conta dispara revoke_from_engine + delete_by_user automaticamente
  • PII nunca em logs — apenas IDs de usuario e milvus_id
