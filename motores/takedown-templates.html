Kubex — Templates de Takedown ​

Templates automatizados para remocao de conteudo infrator.

Implementacao:

  • Templates em texto plain: app/templates/takedown/dmca_en.txt e app/templates/takedown/marco_civil_pt.txt
  • Logica de geracao: app/engines/takedown_engine.py — TakedownEngine.generate()
  • Coleta de evidencias: app/services/evidence_service.py — modo dual (httpx em dev, Playwright em prod)
  • Verificacao automatica: app/tasks/takedown_tasks.py — Celery Beat (verify_takedown_removal)
  • Contatos de abuse: app/engines/takedown_engine.py — get_abuse_contact(url) (20+ plataformas)

Coleta de evidencia com Scrapling / httpx ​

Implementado em: app/services/evidence_service.py — modo dual por ambiente:

  • Dev (evidence_engine=httpx): httpx fetch + hash SHA-256 do HTML (sem screenshot)
  • Prod (evidence_engine=playwright): Playwright headless + screenshot + hash (configuravel no painel admin)

O modo ativo e controlado por platform_settings.evidence_engine (configuravel pelo admin sem redeploy). O storage tambem e dual: storage_provider=local (dev, /tmp) ou storage_provider=r2 (prod, Cloudflare R2).

python
# app/services/evidence_service.py (resumo)

class EvidenceService:
    """Coleta evidencias de conteudo infrator.

    Modo dual controlado por platform_settings.evidence_engine:
      - "httpx"       -> rapido, sem screenshot, ideal para dev
      - "playwright"  -> screenshot real + hash HTML, ideal para prod
    """

    async def collect(self, match_url: str, user_id: int) -> Evidence:
        """Captura screenshot e metadados como prova."""

        settings = await get_cached_settings()

        if settings.evidence_engine == "playwright":
            screenshot_bytes, html_content = await self._collect_playwright(match_url)
        else:
            html_content = await self._collect_httpx(match_url)
            screenshot_bytes = None

        # Hash do HTML como prova de conteudo
        html_hash = hashlib.sha256((html_content or "").encode()).hexdigest()

        evidence = Evidence(
            url=match_url,
            screenshot=screenshot_bytes,
            html_hash=html_hash,
            timestamp=datetime.utcnow(),
            page_title=page.css('title::text').get(),
            domain=extract_domain(match_url),
        )

        # Armazena no R2
        evidence.evidence_url = await upload_to_r2(
            screenshot_bytes,
            f"evidence/{evidence.id}.png"
        )

        return evidence

Verificacao automatica de remocao ​

python
# takedown_verifier.py

from scrapling.fetchers import StealthyFetcher, Fetcher
from celery import shared_task

class TakedownVerifier:
    """Verifica se conteudo foi removido apos takedown."""

    VERIFICATION_SCHEDULE = [
        48 * 3600,      # 48 horas
        7 * 86400,      # 7 dias
        14 * 86400,     # 14 dias
        30 * 86400,     # 30 dias
    ]

    def verify(self, takedown: Takedown) -> VerificationResult:
        """Verifica se a URL ainda esta acessivel."""

        try:
            # Tenta com Fetcher simples primeiro (mais rapido)
            page = Fetcher.get(takedown.target_url, timeout=15)

            if page.status_code in (404, 410, 451):
                return VerificationResult(
                    status="removed",
                    method="http_status",
                    http_code=page.status_code,
                )

            # Se retornou 200, verifica se o conteudo ainda existe
            # (pode ter sido substituido por pagina de "conteudo removido")
            if self._content_removed_indicators(page):
                return VerificationResult(
                    status="removed",
                    method="content_analysis",
                )

            return VerificationResult(
                status="still_online",
                method="content_found",
            )

        except Exception:
            # Se Fetcher falha (Cloudflare), tenta StealthyFetcher
            try:
                page = StealthyFetcher.fetch(
                    takedown.target_url,
                    headless=True,
                    solve_cloudflare=True,
                )

                if self._content_removed_indicators(page):
                    return VerificationResult(status="removed")

                return VerificationResult(status="still_online")

            except Exception:
                return VerificationResult(
                    status="unreachable",
                    method="site_down_or_blocked",
                )

    def _content_removed_indicators(self, page) -> bool:
        """Detecta indicadores de que conteudo foi removido."""
        removal_phrases = [
            # Ingles
            "content has been removed",
            "page not found",
            "this page is no longer available",
            "violates our terms",
            "removed for violating",
            "this content was removed",
            "account has been suspended",
            "this post has been deleted",
            "content removed by moderator",
            "dmca takedown",
            "copyright claim",
            "content unavailable",
            "410 gone",
            # Portugues
            "conteudo removido",
            "pagina nao encontrada",
            "conteudo indisponivel",
            "este post foi removido",
            "violacao dos termos",
            "conta suspensa",
            "conteudo excluido",
        ]
        page_text = page.css('body::text').get('').lower()
        return any(phrase in page_text for phrase in removal_phrases)


@shared_task
def verify_takedown(takedown_id: str):
    """Celery task para verificar remocao."""
    takedown = db.get_takedown(takedown_id)
    verifier = TakedownVerifier()
    result = verifier.verify(takedown)

    db.update_takedown_status(takedown_id, result.status)

    if result.status == "removed":
        notify_creator(takedown.creator_id, f"Conteudo removido: {takedown.target_url}")
    elif result.status == "still_online":
        # Agenda proxima verificacao ou escalona
        next_check = verifier.next_scheduled_check(takedown)
        if next_check:
            verify_takedown.apply_async(args=[takedown_id], countdown=next_check)
        else:
            notify_creator(
                takedown.creator_id,
                f"Conteudo nao removido apos 30 dias: {takedown.target_url}. "
                f"Recomendamos acionar via judicial."
            )
