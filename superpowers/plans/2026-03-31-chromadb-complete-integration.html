ChromaDB Complete Integration — Implementation Plan ​

For agentic workers: REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (- [ ]) syntax for tracking.

Goal: Fix all 8 bugs in ChromaDB integration and bring agent memory system to 100% working state inside Docker.

Architecture: ChromaDB runs as a Docker service (profile memory). KAAS agents use MemoryStore (lazy singleton) to store/search observations via ChromaDB HTTP API. Redis provides query caching. MemoryLifecycleManager handles pruning/consolidation on a Celery Beat schedule.

Tech Stack: ChromaDB 0.4+, Python 3.12, FastAPI, Redis, Celery Beat, pytest, Docker Compose

def _exec_search_memory(tool_input: dict, agent_slug: str = "unknown") -> str:
    """Search the memory store for relevant observations."""
    from app.agents.memory_store import memory_store

    try:
        query = tool_input.get("query", "").strip()
        n_results = min(max(tool_input.get("n_results", 5), 1), 10)

        if not query:
            return "ERROR: Search query cannot be empty."

        memories = memory_store.search(query, agent_slug=agent_slug, n_results=n_results)

        if not memories:
            return "No relevant memories found."

        result = f"Found {len(memories)} relevant memories:\n\n"
        for i, mem in enumerate(memories, 1):
            result += f"{i}. [{mem['metadata'].get('category', 'unknown')}] ({mem['metadata'].get('ts', '?')[:10]})\n"
            result += f"   {mem['content'][:100]}...\n"
            result += f"   (similarity: {1 - mem['distance']:.2f})\n\n"

        return result

    except Exception as e:
        logger.error(f"Error searching memory: {e}")
        return "WARN: Memory search unavailable. (ChromaDB might be offline.)"

import time

# Add to __init__:
    self._last_failure_time = 0
    RETRY_COOLDOWN_SECONDS = 60  # Retry connection every 60s after failure

def _get_client(self):
    """Lazy init ChromaDB client. Returns None if unavailable, retries after cooldown."""
    if self._available is False:
        # Allow retry after cooldown period
        if time.monotonic() - self._last_failure_time < self.RETRY_COOLDOWN_SECONDS:
            return None
        # Reset to allow retry
        self._available = None
        self._client = None

    if self._client is None:
        try:
            import chromadb

            host = os.getenv("CHROMADB_HOST", "localhost")
            if host == "chromadb":
                port = int(os.getenv("CHROMADB_PORT", "8000"))
            else:
                port = int(os.getenv("CHROMADB_PORT", "8001"))

            logger.info(f"Initializing ChromaDB client (host={host}, port={port})")
            self._client = chromadb.HttpClient(host=host, port=port)

            self._collection = self._client.get_or_create_collection(
                name=self.COLLECTION_NAME,
                metadata={"hnsw:space": "cosine"}
            )

            logger.info("ChromaDB client initialized successfully")
            self._available = True

        except Exception as e:
            logger.warning(f"ChromaDB unavailable, continuing without memory store: {e}")
            self._available = False
            self._last_failure_time = time.monotonic()
            return None

    return self._client

class TestChromaDBHealth:
    def test_chromadb_health_available(self):
        """Health endpoint should report ChromaDB status."""
        from app.agents.memory_store import MemoryStore

        store = MemoryStore(redis_client=None)
        store._available = True
        mock_collection = MagicMock()
        mock_collection.count.return_value = 42
        store._collection = mock_collection

        status = {
            "available": store.is_available(),
            "collection": store.COLLECTION_NAME,
            "document_count": mock_collection.count() if store.is_available() else 0,
        }

        assert status["available"] is True
        assert status["document_count"] == 42

    def test_chromadb_health_unavailable(self):
        """Health endpoint should report unavailable gracefully."""
        from app.agents.memory_store import MemoryStore

        store = MemoryStore(redis_client=None)
        store._available = False

        status = {
            "available": store.is_available(),
            "document_count": 0,
        }

        assert status["available"] is False

# Add to existing health/status response
def _get_chromadb_status() -> dict:
    """Get ChromaDB vector store status for health checks."""
    try:
        from app.agents.memory_store import memory_store
        available = memory_store.is_available()
        doc_count = 0
        if available and memory_store._collection:
            try:
                doc_count = memory_store._collection.count()
            except Exception:
                pass
        return {
            "available": available,
            "collection": memory_store.COLLECTION_NAME,
            "document_count": doc_count,
        }
    except Exception:
        return {"available": False, "error": "import failed"}
